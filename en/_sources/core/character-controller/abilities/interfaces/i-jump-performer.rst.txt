IJumpPerformer
==============

.. code-block:: csharp    
    
    namespace Handy2DTools.CharacterController.Abilities
    {
        /// <summary>
        /// Any component that wants to perform jumps 
        /// must implement this Interface.
        /// </summary>
        public interface IJumpPerformer
        {
            void Request();
            void Stop();
            void Perform();
            UnityEvent<GameObject> JumpPerformed { get; }
        }
    }
